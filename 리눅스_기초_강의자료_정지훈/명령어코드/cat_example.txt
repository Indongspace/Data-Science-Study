# <-------------------------------->
# touch로 만든 파일 출력
$ touch newfile
$ cat newfile
$                 <- 아무것도 표시되지 않음

# <-------------------------------->
# /etc/hostname 파일 출력하기
$ cat /etc/hostname

# <-------------------------------->
# /etc/crontab 파일 내용 출력하기
$ cat /etc/crontab

# <-------------------------------->
# 파일 두 개 출력하기
$ cat /etc/hostname /etc/crontab

# <-------------------------------->
# -n 옵션으로 행 번호 표시하기
$ cat -n /etc/crontab

# <-------------------------------->
# 파일을 지정하지 않은 경우, 
# 명령어의 인자로 아무것도 지정하지 않으면 어떻게 될 것인가?
$ cat 
-- 커서가 정지되어 있음. 이 상태에서 cat 명령어를 종료하려면 ctrl + d 입력
